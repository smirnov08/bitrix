```php
<?php 
function plural_form($n, $forms) {
  return $n%10==1&&$n%100!=11?$forms[0]:($n%10>=2&&$n%10<=4&&($n%100<10||$n%100>=20)?$forms[1]:$forms[2]);
}
//usage
$ends = [
   "банан", 
   "банана", 
   "бананов" 
]; 
plural_form(1, $ends); //банан
plural_form(2, $ends); //банана
plural_form(5, $ends); //бананов
```
